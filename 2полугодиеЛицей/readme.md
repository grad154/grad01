from itertools import product
nums = product('01234567', repeat = 5)
k = 0
n1 = '16 36 56 76 61 63 65 67'
n2 = n1.split()
for n1 in nums:
    b = ''.join(n1)
    s = []
    if b.count('6') == 1 and b[0] != '0':
        for x in n2:
            if x in b:
                s.append(x)
        if not s: 
            k+=1
print(k)
